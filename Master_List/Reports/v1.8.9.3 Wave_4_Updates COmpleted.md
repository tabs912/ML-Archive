# **v1.8.9.3.1 Updates Completed \-  Wave 4**

Here is the fully optimized, Wave 4-compliant version of the `createIndexSheet()` function.

### **What was changed to optimize it:**

1. **Unified Batch Painting:** The old function pushed data to the sheet, then ran iterative loops to paint the left side, the right side, and finally used a `RangeList` loop to find and paint the section headers. The new version calculates all background colors, font weights, and font colors simultaneously into three in-memory 2D arrays (`backgrounds`, `fontWeights`, `fontColors`), applying them all at once in a single, massive API push.  
2. **Optimized Grid Clearing:** Replaced the heavy `clearContent().clearFormat().breakApart()` chain with a single `sheet.clear()` command, which handles the exact same teardown in one native backend call.  
3. **Batched Column Resizing:** Replaced the 9-iteration `.forEach()` loop for column widths with two calls to `setColumnWidths()` which resize adjacent column blocks (A-D and F-J) simultaneously.

function removeActiveDemoPPMRsFromDisenrolledExclusion_(demoSheet) {
  if (!demoSheet) return 0;

  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const exclusionSheet = ss.getSheetByName(DISENROLLED_EXCLUSION_SHEET);
  if (!exclusionSheet || exclusionSheet.getLastRow() < DATA_START_ROW) return 0;

  const demoLastRow = demoSheet.getLastRow();
  const demoLastCol = demoSheet.getLastColumn();
  if (demoLastRow < DATA_START_ROW || demoLastCol < 1) return 0;

  const demoHeaders = getHeaders_(demoSheet, HEADER_ROW);
  const demoHeaderMap = buildHeaderIndexMap_(demoHeaders);
  const demoPmrIdx = getPMRIndex_(demoHeaderMap);
  const statusIdx = demoHeaderMap["Enrollment Status"];
  const disenrollEffectiveIdx = findHeaderIndex_(demoHeaderMap, ["Disenrollment Effective Date", "Disenrollment Date", "Date of Death"]);
  if (demoPmrIdx === -1 || statusIdx === undefined) return 0;

  const exclusionData = getDataValues_(exclusionSheet, HEADER_ROW, DATA_START_ROW);
  const exclusionPmrIdx = exclusionData.headerMap ? getPMRIndex_(exclusionData.headerMap) : -1;
  if (exclusionPmrIdx === -1 || !exclusionData.values.length) return 0;

  const exclusionPmrSet = new Set();
  exclusionData.values.forEach(function(row) {
    const pmr = normalizePMR_(row[exclusionPmrIdx]);
    if (pmr) exclusionPmrSet.add(pmr);
  });
  if (!exclusionPmrSet.size) return 0;

  const demoValues = demoSheet.getRange(DATA_START_ROW, 1, demoLastRow - DATA_START_ROW + 1, demoLastCol).getValues();
  const activePMRs = new Set();

  demoValues.forEach(function(row) {
    const pmr = normalizePMR_(row[demoPmrIdx]);
    if (!pmr || !exclusionPmrSet.has(pmr)) return;

    const statusText = String(row[statusIdx] || "").trim().toLowerCase();
    const hasDisenrollmentDate = disenrollEffectiveIdx === -1 ? false : normalizeCompareValue_(row[disenrollEffectiveIdx]) !== "";
    if (statusText === "active" || statusText === "enrolled" || !hasDisenrollmentDate) {
      activePMRs.add(pmr);
    }
  });

  if (!activePMRs.size) return 0;

  // WAVE 4 OPTIMIZATION: Filter in memory and execute a single batch write
  const retainedRows = [];
  let deletedCount = 0;

  exclusionData.values.forEach(function(row) {
    const pmr = normalizePMR_(row[exclusionPmrIdx]);
    if (pmr && activePMRs.has(pmr)) {
      deletedCount++;
    } else {
      retainedRows.push(row);
    }
  });

  if (deletedCount > 0) {
    const width = Math.max(exclusionSheet.getLastColumn(), 1);
    const oldRows = exclusionData.values.length;
    
    // 1. Clear the canvas body instantly
    exclusionSheet.getRange(DATA_START_ROW, 1, oldRows, width).clearContent();
    
    // 2. Repaint only the retained rows
    if (retainedRows.length > 0) {
      const normalizedRows = normalizeRowsToWidth_(retainedRows, width);
      exclusionSheet.getRange(DATA_START_ROW, 1, normalizedRows.length, width).setValues(normalizedRows);
    }
    
    clearSheetRuntimeCachesForSheet_(exclusionSheet);
    logBestEffortWarning_("Re-Enrollment Engine: Surgically purged " + deletedCount + " re-enrolled PMR records from historical Exclusion sheet via in-memory rewrite.");
  }
  return deletedCount;
}

function coerceToValidDate_(value) {
  if (value instanceof Date && !isNaN(value.getTime())) {
    return isReasonableReportDate_(value) ? value : null;
  }

  if (value === null || value === undefined || value === "") return null;

  if (typeof value === "number") {
    if (value < RFF_MIN_SERIAL_DATE) return null;
    return spreadsheetSerialDateToLocalDate_(value);
  }

  const text = String(value).trim();
  if (!text) return null;

  const monthYearMatch = text.match(RFF_RE_MONTH_YEAR);
  if (monthYearMatch) {
    const month = Number(monthYearMatch[1]) - 1;
    let year = Number(monthYearMatch[2]);
    if (year < 100) year += 2000;

    const d = createLocalDateOnly_(year, month, 1);
    if (d.getFullYear() !== year || d.getMonth() !== month) return null;
    return isReasonableReportDate_(d) ? d : null;
  }

  const match = text.match(RFF_RE_DATE_MDY);
  if (match) {
    const month = Number(match[1]) - 1;
    const day = Number(match[2]);
    let year = Number(match[3]);
    if (year < 100) year += 2000;

    const d = new Date(year, month, day);
    if (d.getFullYear() !== year || d.getMonth() !== month || d.getDate() !== day) return null;
    return isReasonableReportDate_(d) ? d : null;
  }

  const d = new Date(text);
  return !isNaN(d.getTime()) && isReasonableReportDate_(d) ? d : null;
}

function appendDisenrolledDeltasToExclusionSheet_(exclusionSheet, sourceHeaders, rowsToAppendByPMR, monthParts, timing) {
  const step = function(label) { if (timing) markFrameworkStep_(timing, "Disenrolled move - exclusion append - " + label); };
  if (!exclusionSheet) return 0;
  updateDisenrolledExclusionReportDates_(exclusionSheet, monthParts);
  if (!rowsToAppendByPMR || rowsToAppendByPMR.size === 0) return 0;

  const ss = typeof exclusionSheet.getParent === "function" ? exclusionSheet.getParent() : SpreadsheetApp.getActive();
  const context = getTemplateDrivenActiveSheetContext_(ss, SHEET_TYPE.DISENROLLED_EXCLUSION);
  const exclusionHeaders = context.headers;
  const sourceHeaderMap = buildHeaderIndexMap_(sourceHeaders || []);
  const sourcePMRIdx = getPMRIndex_(sourceHeaderMap);
  const exclusionHeaderMap = buildHeaderIndexMap_(exclusionHeaders);
  const exclusionPMRIdx = getPMRIndex_(exclusionHeaderMap);
  const exclusionDateIdx = findHeaderIndex_(exclusionHeaderMap, ["Disenrollment Effective Date", "Disenrollment Date", "Date of Death"]);
  const addedIdx = exclusionHeaderMap[DISENROLLED_EXCLUSION_ADDED_HEADER];
  const addedLabel = "Demo P " + (monthParts && monthParts.firstDay ? formatReportDateLabel_(monthParts.firstDay) : "");

  const sourceIndexes = exclusionHeaders.map(function(header) {
    return Object.prototype.hasOwnProperty.call(sourceHeaderMap, header) ? sourceHeaderMap[header] : -1;
  });

  const appendPayload = [];
  rowsToAppendByPMR.forEach(function(rows) {
    (rows || []).forEach(function(sourceRow) {
      if (!sourceRow) return;

      const output = sourceIndexes.map(function(sourceIdx) {
        return sourceIdx === -1 ? "" : sourceRow[sourceIdx];
      });

      if (exclusionPMRIdx >= 0 && sourcePMRIdx >= 0 && !normalizePMR_(output[exclusionPMRIdx])) {
        output[exclusionPMRIdx] = sourceRow[sourcePMRIdx];
      }
      if (addedIdx !== undefined) output[addedIdx] = addedLabel;
      appendPayload.push(output);
    });
  });
  step("append payload built | Rows: " + appendPayload.length);

  if (appendPayload.length === 0) return 0;

  if (exclusionDateIdx !== -1) {
    appendPayload.sort(function(a, b) {
      const dateA = a[exclusionDateIdx] ? new Date(a[exclusionDateIdx]) : new Date(0);
      const dateB = b[exclusionDateIdx] ? new Date(b[exclusionDateIdx]) : new Date(0);
      return dateB - dateA;
    });
    step("append payload sorted newest to oldest");
  }

  const insertStartRow = DATA_START_ROW;
  const width = Math.max(exclusionHeaders.length, 1);

  if (exclusionSheet.getMaxColumns() < width) {
    exclusionSheet.insertColumnsAfter(exclusionSheet.getMaxColumns(), width - exclusionSheet.getMaxColumns());
  }

  exclusionSheet.insertRowsBefore(insertStartRow, appendPayload.length);
  step("rows inserted at top of Disenrolled Exclusion | Rows: " + appendPayload.length);
  
  // OPTIMIZATION: Direct single 2D array write replaces iterative column-run calls
  const targetRange = exclusionSheet.getRange(insertStartRow, 1, appendPayload.length, width);
  targetRange.setValues(appendPayload);
  step("append payload values written | Rows: " + appendPayload.length + "; Width: " + width);

  try {
    const formatSourceRow = insertStartRow + appendPayload.length;
    if (formatSourceRow <= exclusionSheet.getMaxRows()) {
      exclusionSheet.getRange(formatSourceRow, 1, 1, width)
        .copyTo(targetRange, SpreadsheetApp.CopyPasteType.PASTE_FORMAT, false);
      step("append payload formatting copied");
    }
  } catch (err) {
    logBestEffortWarning_("Incremental format brush extension skipped: " + err.message);
  }

  lockFinalOutputRowHeights_(exclusionSheet, "Disenrolled Exclusion");
  step("Disenrolled Exclusion row heights locked");
  clearSheetRuntimeCachesForSheet_(exclusionSheet);
  return appendPayload.length;
}
